// userPrefecences
router.put('/user/preferences', passport.authenticate('jwt', {session: false}), (req, res) =>{
    db.User.findByIdAndUpdate(req.user._id, {
        preferences: req.body.preferences
    })
    .then(user => {
        res.status(201).json(user)
    })
})
